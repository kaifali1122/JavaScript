// VAR - Function scoped, can be re-declared

  var x = 3; // Same variable
}

// LET - Block scoped, cannot be re-declared
let y = 1;
// let y = 2; // ‚ùå Error: Identifier 'y' has already been declared

  let y = 3; // Different variable
}

// CONST - Block scoped, cannot be re-assigned

// z = 2; // ‚ùå Error: Assignment to constant variable
